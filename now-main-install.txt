    sudo dnf install $(cat now-main-install.txt | awk -F': ' '/Имя(\s+)?:(\s+)?/ { print $2 }' | sort -udi)
Adding a package description to a file:
    sudo dnf info $1 | grep 'Имя' 'Репозиторий' 'Из репо' 'Краткое' 'URL' 'Лицензия' 'Описание'
